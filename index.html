history file location

  • cat .bash_history
  • echo $HISTFILESIZE //'maximum history size in file'
  • echo $HISTSIZE //'maximum memory size'
  • history
  • !3 //'specific command from history'
  • !! // 'last command from history'
  • !-3 //'last third command
  • !3:p //'just print third command from history'
  • history -d 191 //'to remove specific command from history'
  • history -c //'clear entire history'
  • echo $HISTCONTROL //'to see history controller
  • $HISTCONTROL=ignoreboth // 'to set not to save duplicate history
  • one space bar + any command //'to not save in history'
  • HISTTIMEFORMAT='%d/%m/%y %T' //'to set show time with history'
  • echo 'HISTTIMEFORMAT="%d/%m/%y %T"' >> .bashrc #to set permanent in RAM

python script mac changer

  • #!/usr/bin/env python3
  • import subprocess
  • interface = 'enp0s3' or input('Interface: ')
  • new_mac = '00:22:44:66:88:00'
  • subprocess.call("ifconfig "+ interface + " down ", shell=True)
  • subprocess.call("ifconfig "+ interface + " hw ether "+ new_mac, shell=True)
  • subprocess.call("ifconfig "+ interface + " up ", shell=True)
